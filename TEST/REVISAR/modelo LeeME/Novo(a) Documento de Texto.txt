import os
import zipfile

def zip_dir(folder_path, zip_path):
    with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
        for root, dirs, files in os.walk(folder_path):
            for file in files:
                file_path = os.path.join(root, file)
                arcname = os.path.relpath(file_path, start=os.path.dirname(folder_path))
                zipf.write(file_path, arcname)
    print(f"Arquivo ZIP criado com sucesso: {zip_path}")

if __name__ == "__main__":
    project_folder = "my_cosmo_project"  # ajuste caso o nome da pasta seja diferente
    output_zip = "my_cosmo_project.zip"
    zip_dir(project_folder, output_zip)
